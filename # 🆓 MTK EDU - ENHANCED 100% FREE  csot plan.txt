```javascript
// Storage routing based on file type and size
const getStorageProvider = (file) => {
  const { size, type, purpose } = file;
  
  // Small profile images (< 1MB) â†’ Supabase
  if (type.includes('image') && size < 1048576) {
    if (purpose === 'profile' || purpose === 'avatar') {
      return 'supabase';
    }
  }
  
  // Optimized images for web â†’ Cloudinary
  if (type.includes('image') && purpose === 'notice_board') {
    return 'cloudinary';
  }
  
  // Documents (PDFs, DOCX) â†’ Backblaze B2
  if (type.includes('pdf') || type.includes('document')) {
    return 'backblaze';
  }
  
  // Large files (> 10MB) â†’ Backblaze B2
  if (size > 10485760) {
    return 'backblaze';
  }
  
  // Certificates, reports â†’ Backblaze B2
  if (purpose === 'certificate' || purpose === 'report') {
    return 'backblaze';
  }
  
  // Default â†’ Supabase for small files
  return 'supabase';
};
```

### **Step 3: Multi-Provider Upload Service**

```php
// app/Services/FileUploadService.php
<?php

namespace App\Services;

use Illuminate\Support\Facades\Storage;

class FileUploadService
{
    public function upload($file, $purpose = 'general')
    {
        $provider = $this->getProvider($file, $purpose);
        
        switch ($provider) {
            case 'supabase':
                return $this->uploadToSupabase($file);
            case 'backblaze':
                return $this->uploadToBackblaze($file);
            case 'cloudinary':
                return $this->uploadToCloudinary($file);
            default:
                return $this->uploadToSupabase($file);
        }
    }
    
    private function getProvider($file, $purpose)
    {
        $size = $file->getSize();
        $mime = $file->getMimeType();
        
        // Small images â†’ Supabase
        if (str_contains($mime, 'image') && $size < 1048576) {
            return 'supabase';
        }
        
        // Notice board â†’ Cloudinary
        if ($purpose === 'notice' || $purpose === 'event') {
            return 'cloudinary';
        }
        
        // Everything else â†’ Backblaze
        return 'backblaze';
    }
    
    private function uploadToBackblaze($file)
    {
        $path = Storage::disk('backblaze')->put('/', $file);
        return [
            'provider' => 'backblaze',
            'path' => $path,
            'url' => Storage::disk('backblaze')->url($path),
        ];
    }
}
```
